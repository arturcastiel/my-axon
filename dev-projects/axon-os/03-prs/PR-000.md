# PR-000 — `my-axon-init` program
project: axon-os
branch:  my-axon-init
status:  not-started
date:    2026-05-07

---

## Purpose

New program that scaffolds a `my-axon/` directory from scratch (fresh mode) or clones an existing private repo (clone mode). Called by boot when `my-axon/` is absent, or directly by the user.

---

## New file: `workspace/programs/my-axon-init.md`

```
# PROGRAM: my-axon-init
# desc:    Initialize or clone the my-axon user-data folder
# usage:   my-axon-init [--mode=fresh | --mode=clone]
# outputs: my-axon/ scaffolded + MYAXON.md written + W:myaxon-* keys set

!NORM | SPAWNED → RUNNING

## IDENTITY LOCK
ASSERT(L:cognition-frame ≡ "AXON-OS") | HALT("Identity lost — run: boot axon")
STORE(W:active-program, "my-axon-init")
LOG(DEBUG, "program-entry: my-axon-init")

## LOAD CONTEXT
ts         ← TOOL(clock)
axon-root  ← RETRIEVE(W:ws-os) | "/home/arturcastiel/opm-hd/axon-project/axon"
# axon-root is parent of axon/ kernel dir — the repo root
axon-repo  ← TOOL(shell, "dirname {axon-root}") | TOOL(shell, "dirname $(dirname {axon-root})")
# my-axon lives as a sibling of axon/ kernel dir, inside the repo root
myaxon-path ← RETRIEVE(W:myaxon-path) | "{TOOL(shell, 'git -C {axon-root} rev-parse --show-toplevel')}/my-axon"
mode       ← RETRIEVE(W:my-axon-init-mode) | "prompt"

## GUARD
IF DIR-EXISTS(myaxon-path) →
  CLEAR(W:active-program)
  FAIL(my-axon-init,
    "my-axon/ already exists at {myaxon-path}.
     To re-initialize: delete it first, then re-run my-axon-init.
     To backup: workspace-backup")

## DISPATCH
IF mode ≡ "fresh" → GOTO(SECTION:FRESH)
IF mode ≡ "clone" → GOTO(SECTION:CLONE)

# Prompt if not specified
→ "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
→ "  AXON — my-axon setup"
→ "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
→ "  my-axon/ stores all your personal data:"
→ "  dev-projects, memory, logs, chats, plans."
→ ""
→ "  Options:"
→ "  [F]  fresh  — create a new empty my-axon/ folder"
→ "  [C]  clone  — restore from your existing my-axon repo on GitHub"
→ "  [S]  skip   — skip for now (limited mode, no data persistence)"
→ "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
answer ← QUERY(user): "F / C / S  (default: F)"
answer ← answer | "F"
IF answer ≡ "F" OR answer ≡ "" → GOTO(SECTION:FRESH)
IF answer ≡ "C"  → GOTO(SECTION:CLONE)
IF answer ≡ "S"  →
  LOG(WARN, "my-axon-init: skipped — running in limited mode")
  CLEAR(W:active-program)
  DONE(my-axon-init)

## SECTION:FRESH
→ ""
→ "  Creating fresh my-axon/ at {myaxon-path} ..."
TOOL(shell, "mkdir -p {myaxon-path}/dev-projects")
TOOL(shell, "mkdir -p {myaxon-path}/memory/longterm")
TOOL(shell, "mkdir -p {myaxon-path}/memory/episodic")
TOOL(shell, "mkdir -p {myaxon-path}/memory/working")
TOOL(shell, "mkdir -p {myaxon-path}/memory/local")
TOOL(shell, "mkdir -p {myaxon-path}/memory/kv-store")
TOOL(shell, "mkdir -p {myaxon-path}/memory/semantic-index")
TOOL(shell, "mkdir -p {myaxon-path}/log/entries")
TOOL(shell, "mkdir -p {myaxon-path}/log/igap")
TOOL(shell, "mkdir -p {myaxon-path}/log/turns")
TOOL(shell, "mkdir -p {myaxon-path}/chats")
TOOL(shell, "mkdir -p {myaxon-path}/plans")
TOOL(shell, "mkdir -p {myaxon-path}/events")
TOOL(shell, "mkdir -p {myaxon-path}/libraries")
TOOL(shell, "mkdir -p {myaxon-path}/generated")
TOOL(shell, "touch {myaxon-path}/chats/INDEX.md")
TOOL(shell, "touch {myaxon-path}/plans/INDEX.md")
TOOL(shell, "touch {myaxon-path}/events/event-log.json")

# Write .gitignore
WRITE("{myaxon-path}/.gitignore",
"# Machine-specific keys — never push
memory/local/
# Runtime caches
memory/kv-store/cache.db
memory/semantic-index/
memory/episodic/prompt-log.jsonl
# Raw library files (PDFs) — text outputs are kept
libraries/*/articles/
libraries/*/*.pdf
libraries/*/*.PDF
# Scratch
memory/working/
")

GOTO(SECTION:WRITE-MYAXON)

## SECTION:CLONE
→ ""
→ "  Cloning my-axon from your GitHub repo."
url ← QUERY(user): "Paste your my-axon repo URL (e.g. https://github.com/you/my-axon):"
ASSERT(url matches /^https?:\/\//) | FAIL(my-axon-init, "Invalid URL. Must start with https://")
→ ""
→ "  Run in your terminal:"
→ ""
→ "    git clone {url} {myaxon-path}"
→ ""
step-done ← QUERY(user): "Clone complete? [yes / error]"
IF step-done ≢ "yes" →
  CLEAR(W:active-program)
  HALT("my-axon-init paused at clone step. Re-run after cloning.")
GOTO(SECTION:WRITE-MYAXON)

## SECTION:WRITE-MYAXON
# Get user name
user-name ← RETRIEVE(W:myaxon-name) | QUERY(user): "Your name for this workspace? (e.g. arturcastiel)"

WRITE("{myaxon-path}/MYAXON.md",
"# MYAXON
> Personal user-data folder for AXON. Private — back up to your own repo.
> OS lives separately at: {axon-root}

---

## CONFIGURATION

\`\`\`
Name:    {user-name}
Created: {ts.date}
OS:      {axon-root}
\`\`\`

---

## PATH MAP

\`\`\`
STORE(W:myaxon-name,         \"{user-name}\")
STORE(W:myaxon-path,         \"{myaxon-path}\")
STORE(W:myaxon-dev-projects, \"{myaxon-path}/dev-projects/\")
STORE(W:myaxon-memory,       \"{myaxon-path}/memory/\")
STORE(W:myaxon-longterm,     \"{myaxon-path}/memory/longterm/\")
STORE(W:myaxon-episodic,     \"{myaxon-path}/memory/episodic/\")
STORE(W:myaxon-working,      \"{myaxon-path}/memory/working/\")
STORE(W:myaxon-local,        \"{myaxon-path}/memory/local/\")
STORE(W:myaxon-log,          \"{myaxon-path}/log/entries/\")
STORE(W:myaxon-igap,         \"{myaxon-path}/log/igap/\")
STORE(W:myaxon-turns,        \"{myaxon-path}/log/turns/\")
STORE(W:myaxon-chats,        \"{myaxon-path}/chats/\")
STORE(W:myaxon-plans,        \"{myaxon-path}/plans/\")
STORE(W:myaxon-libraries,    \"{myaxon-path}/libraries/\")
STORE(W:myaxon-generated,    \"{myaxon-path}/generated/\")
\`\`\`
")

# Store keys in session
STORE(W:myaxon-name,         user-name)
STORE(W:myaxon-path,         myaxon-path)
STORE(W:myaxon-dev-projects, "{myaxon-path}/dev-projects/")
STORE(W:myaxon-memory,       "{myaxon-path}/memory/")
STORE(W:myaxon-longterm,     "{myaxon-path}/memory/longterm/")
STORE(W:myaxon-episodic,     "{myaxon-path}/memory/episodic/")
STORE(W:myaxon-working,      "{myaxon-path}/memory/working/")
STORE(W:myaxon-local,        "{myaxon-path}/memory/local/")
STORE(W:myaxon-log,          "{myaxon-path}/log/entries/")
STORE(W:myaxon-igap,         "{myaxon-path}/log/igap/")
STORE(W:myaxon-turns,        "{myaxon-path}/log/turns/")
STORE(W:myaxon-chats,        "{myaxon-path}/chats/")
STORE(W:myaxon-plans,        "{myaxon-path}/plans/")
STORE(W:myaxon-libraries,    "{myaxon-path}/libraries/")
STORE(W:myaxon-generated,    "{myaxon-path}/generated/")

LOG(INFO, "my-axon-init: scaffold complete at {myaxon-path}")

→ ""
→ "  ✓ my-axon/ created at {myaxon-path}"
→ "  ✓ MYAXON.md written with absolute paths"
→ "  ✓ W:myaxon-* keys loaded for this session"
→ ""
→ "  Next: run workspace-backup setup  to link to your GitHub repo."

CLEAR(W:active-program)
DONE(my-axon-init)
```
